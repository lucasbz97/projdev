CREATE DATABASE PROJDEV;

USE PROJDEV;

CREATE TABLE USUARIO(
ID_USUA INT NOT NULL AUTO_INCREMENT, 
NOME_USUA VARCHAR(35) NOT NULL, 
EMAIL_USUA VARCHAR(60) NOT NULL,
SENHA_USUA VARCHAR(30) NOT NULL
,PRIMARY KEY(ID_USUA));

CREATE TABLE SALA(
ID_SALA INT NOT NULL AUTO_INCREMENT,
ID_USUA INT NOT NULL,
NOME_SALA VARCHAR(60) NOT NULL,
PK(ID_SALA),
FK(ID_USUA));

CREATE TABLE PERGUNTAS(
ID_PERG INT NOT NULL AUTO_INCREMENT,
ID_SALA INT NOT NULL,
QUESTAO_PERG VARCHAR(60) NOT NULL,
PRIMARY KEY (ID_PERG)
FK(ID_SALA));

CREATE TABLE RESPOSTAS(
ID_RESP INT NOT NULL AUTO_INCREMENT,
ID_PERG INT NOT NULL,
ALTERNATIVA_RESP VARCHAR(60),
CHECKED_RESP TINYINT(1) NOT NULL,
PRIMARY KEY(ID_RESP),
FK(ID_PERG));
